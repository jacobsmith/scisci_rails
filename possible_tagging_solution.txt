Query based on something akin to `where :project = @project`

class Note
  has_many :note_tags
  has_many :tags, through: :note_tags
end

class Tag
  has_many :note_tags
  has_many :notes, :through :note_tags
end

class NoteTag
  belongs_to :note
  belongs_to :tag
end
